import javafx.util.Pair;

import java.io.File;
import java.io.FileNotFoundException;
import java.util.*;

public class Main{

    public static void main(String[] args)throws FileNotFoundException{
        //System.out.println(new Solution().func());
        new Solution().func();
    }
    static class Solution {
        int a,b,c,n;
        int[] map,vis;
        int ans;
        void func() throws FileNotFoundException {
            //Scanner in= new Scanner(System.in);
            Scanner in= new Scanner(new File("C:\\Users\\lingfly\\IdeaProjects\\first\\src\\in.txt"));
            n=in.nextInt();
            a=in.nextInt();
            b=in.nextInt();
            c=in.nextInt();
            ans=Integer.MAX_VALUE;
            map=new int[n+10];
            vis=new int[n+10];
            for (int i=1;i<=n;i++){

                map[i]=in.nextInt();
            }
            dfs(1,0);
            System.out.println(ans);

        }
        void dfs(int cur,int dept){
            if (cur<1||cur>n||vis[cur]==1)return;
            if (cur==n){
                ans=Math.min(ans,dept);
                return;
            }
            vis[cur]=1;
            dfs(map[cur],dept+a);
            vis[cur]=0;
            for (int i=1;i<n;i++){
                map[cur]--;
                vis[cur]=1;
                dfs(map[cur],dept+a+b);
                vis[cur]=0;
                map[cur]++;
            }
            for (int i=1;i<n;i++){
                map[cur]++;
                vis[cur]=1;
                dfs(map[cur],dept+a+c);
                vis[cur]=0;
                map[cur]--;
            }
        }
    }

}
